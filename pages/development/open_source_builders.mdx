import OpenSourceBuildersTable from "../../components/tables/OpenSourceBuildersTable";
import openSourceBuildersAPI from "../../scripts/API/openSourceBuildersAPI";
import openSourceBuildersData from "../../data/builders-demo-improved.json";

export const getStaticProps = async () => {
  try {
    const allRepos = await openSourceBuildersAPI(openSourceBuildersData);
    
    // Filter out completely failed requests (those with only error data)
    const validRepos = allRepos.filter(repo => 
      repo.starCount !== undefined || repo.mostRecentRepo || repo.mostStarredRepo
    );
    
    // If we have some valid data, use it; otherwise fall back to basic data
    const dataToUse = validRepos.length > 0 ? allRepos : openSourceBuildersData.map((team, index) => ({
      ...team,
      key: `fallback-${index}-${team.website}`,
      starCount: 0,
      repos: [],
      repoCount: 0,
      mostRecentRepo: null,
      mostStarredRepo: null,
      reposOnGithub: team.teamGithubURL,
      error: "GitHub API unavailable",
    }));
    
    const sortedOSBuilderData = dataToUse.sort((a, b) => {
      const dateA = new Date(a.mostRecentRepo?.pushedAt || 0).getTime();
      const dateB = new Date(b.mostRecentRepo?.pushedAt || 0).getTime();
      return dateB - dateA;
    });

    return {
      props: { sortedOSBuilderData },
      revalidate: 7200, // 2 hours for production stability
    };

} catch (error) {
console.error('Failed to fetch open source builders data:', error);

// Always provide fallback data instead of notFound
const fallbackData = openSourceBuildersData.map((team, index) => ({
  ...team,
  key: `error-fallback-${index}-${team.website}`,
  starCount: 0,
  repos: [],
  repoCount: 0,
  mostRecentRepo: null,
  mostStarredRepo: null,
  reposOnGithub: team.teamGithubURL,
  error: "Service temporarily unavailable",
}));

return {
props: { sortedOSBuilderData: fallbackData },
revalidate: 60, // Try again sooner when in error state
};
}
};

export default function OpenSourceBuilders({ sortedOSBuilderData }) {
  return (
    <>
      <h1 className="_mt-2 _text-4xl _font-bold _tracking-tight _text-slate-900 dark:_text-slate-100">
        Open Source Builders
      </h1>
      <p className="_mt-6 _leading-7 first:_mt-0">
        Explore teams building open-source dApps and tools on Cardano.
      </p>
      <br />
      <OpenSourceBuildersTable data={sortedOSBuilderData} />
    </>
  );
}
